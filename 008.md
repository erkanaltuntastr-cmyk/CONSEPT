// core/storage/localStorageAdapter.ts
export function createLocalStorageAdapter<T>(key: string) {
  return {
    load(): T | null {
      const raw = localStorage.getItem(key);
      if (!raw) return null;
      try {
        return JSON.parse(raw) as T;
      } catch {
        return null;
      }
    },
    save(value: T) {
      localStorage.setItem(key, JSON.stringify(value));
    },
  };
}

export function addSetToMovement(
  day: WorkoutDay,
  movementId: string,
  input: { weight: number; reps: number; rir?: number; note?: string }
): WorkoutDay {
  const set: WorkoutSet = {
    setId: crypto.randomUUID(),
    movementId,
    weight: input.weight,
    reps: input.reps,
    rir: input.rir,
    note: input.note,
  };
  return {
    ...day,
    completedSets: [...day.completedSets, set],
  };
}

        // Basic migration example
        if (version < 2 && state) {
          // Add defaults or transform structures if needed
        }
        return state as any;
      },

export function createDailyWorkoutState(programDay: { id: string; movements: { movementId: string }[] }): WorkoutState {

export function addSet(state: WorkoutState, movementId: string, setData: { reps: number; weight?: number; rpe?: number }): WorkoutState {
  return {
    ...state,
    movements: state.movements.map((m) =>
      m.movementId === movementId ? { ...m, sets: [...m.sets, { ...setData, ts: new Date().toISOString() }] } : m
    ),
  };
}

export function removeLastSet(state: WorkoutState, movementId: string): WorkoutState {
  return {
    ...state,
    movements: state.movements.map((m) =>
      m.movementId === movementId ? { ...m, sets: m.sets.slice(0, -1) } : m
    ),
  };
}

import lunr from 'lunr';
import type { Movement } from '../models/movement';

let index: lunr.Index | null = null;

export async function buildIndex(movements: Movement[]) {
  index = lunr(function () {
    this.ref('id');
    this.field('name');
    this.field('tags');
    movements.forEach((m) => this.add(m));
  });
}

export function searchMovements(query: string, movements: Movement[]) {
  // Lightweight filter first
  const filtered = query
    ? movements.filter((m) => m.name.toLowerCase().includes(query.toLowerCase()))
    : movements;

  if (!index) return filtered;
  try {
    const results = index.search(query);
    const ids = new Set(results.map((r) => r.ref));
    return filtered.filter((m) => ids.has(m.id));
  } catch {
    return filtered;
  }
}

// Toast.tsx
import { useEffect, useState } from 'react';
export function Toast({ message }: { message: string }) {
  const [show, setShow] = useState(true);
  useEffect(() => { const t = setTimeout(() => setShow(false), 2000); return () => clearTimeout(t); }, []);
  if (!show) return null;
  return <div className="fixed bottom-4 left-1/2 -translate-x-1/2 bg-primary text-white px-4 py-2 rounded">{message}</div>;
}

// useToast.ts
import { useState } from 'react';
export function useToast() {
  const [msg, setMsg] = useState<string | null>(null);
  const toast = (o: { title: string; description?: string; type?: 'success' | 'info' | 'warning' }) => {
    setMsg(`${o.title}${o.description ? ` — ${o.description}` : ''}`);
    setTimeout(() => setMsg(null), 2200);
  };
  const ToastOutlet = () => (msg ? <Toast message={msg} /> : null);
  return { toast, ToastOutlet };
}

import type { WorkoutState } from "@/core/data/models/program";

export function createDailyWorkoutState(args: {
  programDayId: string;
  movementIds: string[];
  dateISO: string;
}): WorkoutState {
  return {
    dayId: `${args.programDayId}-${args.dateISO}`,
    programDayId: args.programDayId,
    date: args.dateISO,
    movements: args.movementIds.map((movementId) =&gt; ({
      movementId,
      sets: [],
    })),
  };
}

export function addSet(
  state: WorkoutState,
  movementId: string,
  setData: { reps: number; weight?: number; rpe?: number }
): WorkoutState {
  return {
    ...state,
    movements: state.movements.map((m) =&gt;
      m.movementId === movementId
        ? {
            ...m,
            sets: [
              ...m.sets,
              { ...setData, ts: new Date().toISOString() },
            ],
          }
        : m
    ),
  };
}

export function removeLastSet(
  state: WorkoutState,
  movementId: string
): WorkoutState {
  return {
    ...state,
    movements: state.movements.map((m) =&gt;
      m.movementId === movementId
        ? { ...m, sets: m.sets.slice(0, -1) }
        : m
    ),
  };
}

export function getTotalVolume(state: WorkoutState): number {
  return state.movements.reduce((total, m) =&gt; {
    const movementVolume = m.sets.reduce((acc, s) =&gt; {
      if (!s.weight) return acc;
      return acc + s.weight * s.reps;
    }, 0);
    return total + movementVolume;
  }, 0);
}

Oestv2 Readme

Oakwood Easy Sport Tracker V2 (OESTV2)
OESTV2 is a minimal, fast, and fully local-first workout tracker designed to deliver a clean, distraction-free training experience. The app is built with a modern, modular architecture using React, Vite, TypeScript, Zustand, Zod, and Tailwind CSS.

🚀 Features (MVP)
Local-first architecture — no login, no cloud dependency

Onboarding Flow — basic profile, goals, preferences

Workout Engine

Movement list

Add / undo sets

Automatic volume calculation

Progress Overview — simple training statistics

AI Support (optional)

Daily status classification

Session modification advice

Modular structure — easy to extend

PWA-ready — offline capability prepared for future release

🧱 Architecture Overview
src/
  app/              # Router, layout, global error boundary
  core/             # Domain logic: models, engine, movement library, AI
  features/         # Onboarding, Workout, Progress modules
  shared/           # UI system, reusable components, hooks, utils
  store/            # Global Zustand store with persist + migrations
  assets/           # Icons, graphics, illustrations
Layer Summary
core/ → Pure logic (program generator, workout engine, Zod schemas)

features/ → Each user-facing capability as a self-contained module

shared/ → Visual components, design tokens, utility hooks

store/ → Global state: program, workout, profile, settings, onboarding flag

🛠 Installation
1. Clone the repository
git clone https://github.com/erkanaltuntastr-cmyk/OESTV2.git
2. Navigate into the project folder
cd OESTV2
3. Install dependencies
npm install
4. Start the development server
npm run dev
Your local app will run at:

http://localhost:5173
🧠 AI Integration (Optional)
AI is not required for core functionality. All AI requests are routed through a secure backend proxy:

Prompts: /src/core/ai/prompts.ts

Client service: /src/core/ai/aiService.ts

Endpoint: POST /api/ai

No API keys are exposed on the client.
