import { useNavigate } from "react-router-dom";
import { programData } from "../../data/initialProgram";

export default function Welcome({ user, lastSession }) {
  const navigate = useNavigate();

  const today = new Date().toLocaleString("en-GB", {
    weekday: "long",
    day: "numeric",
    month: "long",
  });

  // Varsayılan gün – kullanıcı son gün seçtiyse onu gösterelim
  const defaultDay = lastSession?.workoutId || "day1";

  const todaysWorkout = programData[defaultDay]?.exercises || [];

  return (
    <div className="min-h-screen bg-oakwood-bg text-oakwood-dark px-6 py-10">

      {/* HEADER */}
      <header className="flex items-center justify-between mb-10">
        <h1 className="font-display text-3xl font-extrabold tracking-tight text-oakwood-dark">
          OAKWOOD
        </h1>
        <p className="text-xs font-bold tracking-[0.3em] text-oakwood-gray uppercase">
          Easy Sport Tracker
        </p>
      </header>

      {/* WELCOME TEXT */}
      <div className="mb-10 text-center">
        <h2 className="text-3xl font-extrabold text-oakwood-dark">
          Welcome, {user || "Athlete"}!
        </h2>
        <p className="text-sm text-oakwood-gray mt-2">{today}</p>
      </div>

      {/* LAST SESSION */}
      {lastSession && (
        <div className="bg-white rounded-2xl p-6 shadow-md border border-oakwood-border mb-8">
          <h3 className="text-lg font-bold text-oakwood-dark text-center mb-4">
            Where Did We Leave Off?
          </h3>
          <div className="space-y-2 text-sm">
            <div className="flex justify-between">
              <span className="text-oakwood-gray">Last Workout</span>
              <span className="font-semibold">
                {lastSession.workoutTitle}
              </span>
            </div>
            <div className="flex justify-between">
              <span className="text-oakwood-gray">Date</span>
              <span className="font-semibold">{lastSession.date}</span>
            </div>
          </div>
        </div>
      )}

      {/* TODAY'S WORKOUT */}
      <div className="bg-white rounded-2xl p-6 shadow-md border border-oakwood-border mb-8">
        <h3 className="text-lg font-bold text-oakwood-dark text-center mb-5">
          Today's Workout
        </h3>

        {/* WORKOUT LIST */}
        <div className="space-y-3 text-sm">
          {todaysWorkout.map((ex, index) => (
            <div
              key={ex.id}
              className="flex items-center justify-between border-b border-oakwood-border pb-3"
            >
              {/* INDEX */}
              <span className="text-oakwood-gray font-bold text-xs">
                {ex.id.startsWith("W") ? (
                  <span className="px-2 py-1 rounded-md text-[10px] bg-green-100 text-green-700">
                    {ex.id}
                  </span>
                ) : (
                  <span className="px-2 py-1 rounded-md text-[10px] bg-gray-200 text-gray-700">
                    {ex.id}
                  </span>
                )}
              </span>

              {/* NAME */}
              <span className="flex-1 text-center font-medium">
                {index + 1}. {ex.name}
              </span>
            </div>
          ))}
        </div>

        {/* Start Button */}
        <button
          onClick={() => navigate(/tracker/${defaultDay})}
          className="w-full mt-6 py-3 bg-oakwood-light text-white font-bold rounded-xl shadow-md hover:bg-oakwood-dark transition"
        >
          Start Workout
        </button>
      </div>

      {/* CONTACT BUTTON */}
      <div className="text-center">
        <button
          onClick={() => navigate("/contact")}
          className="text-sm text-oakwood-gray underline hover:text-oakwood-dark"
        >
          Contact / Support
        </button>
      </div>
    </div>
  );
}

🔥 OAKWOOD – AI CONTEXTUAL GUIDANCE PROMPT (PROD-READY, FINAL v1.0)
System Prompt (AI için)
You are an assistant that generates soft, low-visibility guidance text for a fitness onboarding form.
Your task is not to give actual fitness advice, instructions, judgments, or prescriptive recommendations.
Instead, you produce natural, guess-based example texts that a user might say, based on their profile.
